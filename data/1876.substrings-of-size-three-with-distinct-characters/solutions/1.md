class Solution {
    public int countGoodSubstrings(String s) {
        int l=s.length();
        String c="";
        String cc="";
        int lm=0;
        int res=0;
        int rm=lm+3;
        while(rm<=l){
            c=s.substring(lm,rm);
            for(int i=0;i<c.length();i++){
                if(i==0) cc = c.charAt(i)+"";
                else if(cc.contains(c.charAt(i)+"")){
                    break;
                }else{
                    cc += c.charAt(i)+"";
                    continue;
                }
            }
            if(cc.length() == 3)  res++; 
            cc="";
            lm++;
            rm++;
        }
        return res;
    }
}
