```python3 []
class Solution:
    def divideString(self, s: str, k: int, fill: str) -> List[str]:
        n = len(s)
        
        res = []
        for i in range(0, n, k):
            if i + k <= n:
                res.append(s[i : i+k])
            else:
                res.append(s[i : n] + fill * (k - (n - i)))
        
        return res
```

```c++ []
class Solution 
{
public:
    vector<string> divideString(string s, int k, char fill) 
    {
        int n = (int)s.size();

        vector<string> res;
        for (int i = 0; i < n; i += k)
        {
            if (i + k <= n)
            {
                res.push_back(s.substr(i, k));
            }
            else 
            {
                string cur = s.substr(i, (n - i)) + string(k - (n - i), fill);
                res.push_back(cur);
            }
        }

        return res;

    }
};
```

```java []
class Solution 
{
    public String[] divideString(String s, int k, char fill) 
    {
        int n = s.length();

        int rn = (int)Math.ceil((double)n / k);
        String [] res = new String[rn];

        int ri = 0;
        for (int i = 0; i < n; i += k)
        {
            if (i + k <= n)
            {
                res[ri ++] = s.substring(i, i + k); 
            }
            else
            {
                res[ri ++] = s.substring(i, n) + Character.toString(fill).repeat(k - (n - i));
            }
        }

        return res;
    }
}
```
