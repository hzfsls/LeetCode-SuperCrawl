```python3 []
class Solution:
    def getModifiedArray(self, length: int, updates: List[List[int]]) -> List[int]:
        n = length
        ################ 典型差分，上下台阶的思想，"出租车上下车问题"
        f = [0 for _ in range(n)]   #差分数组
        #### 差分
        for start, end, diff in updates:
            f[start] += diff
            if end + 1 < n:
                f[end + 1] -= diff
        #### 整理
        for i in range(1, n):
            f[i] += f[i-1]
        return f
```
```c++ []
class Solution
 {
public:
    vector<int> getModifiedArray(int length, vector<vector<int>>& updates) 
    {
        int n = length;
        //////////////// 典型差分，上下台阶的思想，出租车上下车问题
        vector<int> f(n, 0);
        //// 差分标记
        for (auto u: updates)
        {
            int start = u[0],   end = u[1],     diff = u[2];
            f[start] += diff;
            if (end + 1 < n)
            f[end + 1] -= diff;
        }
        ////整理
        for (int i = 1; i < n; i ++)
            f[i] += f[i-1];
        
        return f;
    }
};
```
